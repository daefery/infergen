## What This Epic Delivers

`infergen view` subcommand generates a self-contained HTML file that displays the event catalog as an interactive, searchable web UI. Fully offline — no HTTP server, no external network calls. All catalog data is embedded as JSON in a single `catalog-viewer.html` file that opens in the default browser.

**Features:**
- Stats bar: total / approved / proposed / ignored counts
- Search box: live filter by event name (case-insensitive substring)
- Filter by status (all / proposed / approved / ignored)
- Filter by event kind (pageView / apiCall / …)
- PII-only toggle (show only events with ≥ 1 PII property)
- Event table: Name, Kind, Status, Confidence (visual bar), PII flag, source count
- Click row to expand: description, all properties (name + type + PII badge), all provenances (path + line), providers

---

## Open Questions / Risks

- **`open` crate availability**: `open` crate v5 is cross-platform (macOS/Linux/Windows). Falls back gracefully — prints file path on open failure.
- **Large catalogs**: Embedding all data as JSON in HTML is fine for typical catalog sizes (< 10k events). No pagination needed in v1.
- **CI environments**: `--no-open` flag prevents auto-opening browser in CI.

const CATALOG = __CATALOG_JSON__;
const events = CATALOG.events || [];
let expandedId = null;

function statusClass(s) {
  return s === 'approved' ? 'st-approved' : s === 'ignored' ? 'st-ignored' : 'st-proposed';
}
function confClass(c) {
  return c >= 0.8 ? 'conf-hi' : c >= 0.5 ? 'conf-mid' : 'conf-lo';
}
function hasPii(ev) {
  return (ev.properties || []).some(p => p.pii);
}
function kindLabel(k) {
  const m = {pageView:'pageView',apiCall:'apiCall',authEvent:'authEvent',
    formSubmit:'formSubmit',buttonClick:'buttonClick',search:'search',error:'error'};
  return m[k] || k;
}
function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function renderStats() {
  const tot = events.length;
  const app = events.filter(e=>e.status==='approved').length;
  const pro = events.filter(e=>e.status==='proposed').length;
  const ign = events.filter(e=>e.status==='ignored').length;
  document.getElementById('stats').innerHTML =
    `<span class="badge badge-total">${tot} total</span>` +
    `<span class="badge badge-approved">${app} approved</span>` +
    `<span class="badge badge-proposed">${pro} proposed</span>` +
    `<span class="badge badge-ignored">${ign} ignored</span>`;
}

function buildDetail(ev) {
  const props = (ev.properties||[]);
  const provs = (ev.provenance||[]);
  const provids = (ev.providers||[]);
  const descHtml = ev.description
    ? `<p class="desc">${esc(ev.description)}</p>`
    : `<p class="no-desc">(no description)</p>`;
  const propsHtml = props.length
    ? `<ul class="prop-list">${props.map(p =>
        `<li><span class="prop-name">${esc(p.name)}</span>`+
        `<span class="prop-type">${esc(p.type||'unknown')}</span>`+
        (p.pii?`<span class="pii-yes" title="PII">🔒</span>`:'')+
        (p.required?`<span class="mono" style="color:#94a3b8">required</span>`:'')+
        `</li>`).join('')}</ul>`
    : `<span style="color:#475569;font-size:.8rem">none</span>`;
  const provsHtml = provs.length
    ? `<ul class="prov-list">${provs.map(p =>
        `<li>${esc(p.sourcePath)}${p.line!=null?' :'+p.line:''}${p.adapter?' ('+esc(p.adapter)+')':''}</li>`
      ).join('')}</ul>`
    : `<span style="color:#475569;font-size:.8rem">none</span>`;
  const providersHtml = provids.length
    ? provids.map(p=>`<span class="provider-chip">${esc(p)}</span>`).join('')
    : `<span style="color:#475569;font-size:.8rem">none configured</span>`;
  return `<td colspan="6" class="detail">
    <div class="detail-section">${descHtml}</div>
    <div class="detail-grid">
      <div>
        <h3>Properties (${props.length})</h3>${propsHtml}
      </div>
      <div>
        <h3>Source Locations (${provs.length})</h3>${provsHtml}
        <h3 style="margin-top:.75rem">Providers</h3>${providersHtml}
        <h3 style="margin-top:.75rem">ID</h3>
        <span class="mono">${esc(ev.id)}</span>
      </div>
    </div>
  </td>`;
}

function render() {
  const q = document.getElementById('search').value.toLowerCase();
  const st = document.getElementById('status-filter').value;
  const ki = document.getElementById('kind-filter').value;
  const piiOnly = document.getElementById('pii-filter').checked;

  const filtered = events.filter(ev => {
    if (q && !ev.name.toLowerCase().includes(q)) return false;
    if (st && ev.status !== st) return false;
    if (ki && ev.kind !== ki) return false;
    if (piiOnly && !hasPii(ev)) return false;
    return true;
  });

  document.getElementById('count-note').textContent =
    filtered.length === events.length
      ? `${events.length} events`
      : `${filtered.length} of ${events.length} events`;

  const tbody = document.getElementById('tbody');
  tbody.innerHTML = '';
  document.getElementById('empty-state').style.display = filtered.length ? 'none' : 'block';

  filtered.forEach(ev => {
    const pct = Math.round(ev.confidence * 100);
    const srcs = (ev.provenance||[]).length;
    const piiFlag = hasPii(ev);
    const row = document.createElement('tr');
    row.dataset.id = ev.id;
    row.innerHTML =
      `<td class="name-cell">${esc(ev.name)}</td>`+
      `<td><span class="kind-badge">${esc(kindLabel(ev.kind))}</span></td>`+
      `<td><span class="st ${statusClass(ev.status)}">${esc(ev.status)}</span></td>`+
      `<td><div class="conf-wrap"><div class="conf-bar"><div class="conf-fill ${confClass(ev.confidence)}" style="width:${pct}%"></div></div><span class="conf-pct">${pct}%</span></div></td>`+
      `<td>${piiFlag?'<span class="pii-yes" title="Has PII properties">🔒</span>':'<span class="pii-no">–</span>'}</td>`+
      `<td class="mono">${srcs}</td>`;
    row.addEventListener('click', () => {
      if (expandedId === ev.id) {
        expandedId = null;
        const det = document.getElementById('detail-'+ev.id);
        if (det) det.parentElement.removeChild(det);
      } else {
        if (expandedId) {
          const prev = document.getElementById('detail-'+expandedId);
          if (prev) prev.parentElement.removeChild(prev);
        }
        expandedId = ev.id;
        const detRow = document.createElement('tr');
        detRow.id = 'detail-'+ev.id;
        detRow.innerHTML = buildDetail(ev);
        row.parentElement.insertBefore(detRow, row.nextSibling);
      }
    });
    tbody.appendChild(row);
  });
}

renderStats();
render();
</script>
</body>
</html>

```rust
const TEMPLATE: &str = include_str!("view.html");

pub fn run(args: ViewArgs) -> anyhow::Result<()> {
    // 1. Load catalog (return empty catalog if file doesn't exist — still generate viewer)
    let catalog = match load_catalog(&args.catalog) {
        Ok(c) => c,
        Err(infergen_core::Error::Io(e)) if e.kind() == std::io::ErrorKind::NotFound => {
            eprintln!("note: catalog not found at {} — generating empty viewer", args.catalog.display());
            Catalog::default()
        }
        Err(e) => return Err(e.into()),
    };

    // 2. Serialize catalog to JSON
    let catalog_json = serde_json::to_string(&catalog)
        .context("serialize catalog to JSON")?;

    // 3. Inject into template
    let html = TEMPLATE.replace("__CATALOG_JSON__", &catalog_json);

    // 4. Determine output path
    let output = match &args.output {
        Some(p) => p.clone(),
        None => args.catalog
            .parent()
            .unwrap_or_else(|| std::path::Path::new("."))
            .join("catalog-viewer.html"),
    };

    // 5. Write HTML file (create parent dirs)
    if let Some(parent) = output.parent()
        && !parent.as_os_str().is_empty()
    {
        std::fs::create_dir_all(parent)
            .with_context(|| format!("create dir {}", parent.display()))?;
    }
    std::fs::write(&output, html)
        .with_context(|| format!("write viewer to {}", output.display()))?;

    println!("catalog viewer written to {}", output.display());

    // 6. Open browser (unless --no-open)
    if !args.no_open {
        if let Err(e) = open::that(&output) {
            eprintln!("note: could not open browser: {e}");
            eprintln!("      open manually: {}", output.display());
        }
    } else {
        println!("open in browser: {}", output.display());
    }

    Ok(())
}
```

**Edge cases:**
- Catalog not found → use empty `Catalog::default()`, still write viewer (shows 0 events + helpful stats)
- Output dir doesn't exist → `create_dir_all` creates it
- Browser open failure → `eprintln!` and print path, never hard-fail
- `--output` provided → write to custom path
- `catalog.parent()` is None (catalog = just filename, no dir) → fallback to `"."` avoids panic

**Dependencies:** `infergen_core::{Catalog, load_catalog}`, `serde_json`, `open`, `anyhow`
